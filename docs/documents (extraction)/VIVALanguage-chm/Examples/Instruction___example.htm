Example 1

 

DECLARE CHANNEL WR=56

DECLARE CHANNEL RD=86

TIMING TIM1;

PERIOD 4C;

STROBE 3C;

ENDTIMING;

START PRG1;

/ TIMING TIM1;

IL WR;

IH RD;

/; !Pattern 1

IH WR;

IL; RD;

/; !Pattern 2

ENDTEST

EXAMPLE 2

ENDTIMING;

IH WR;

IL; RD;

PATTERN; !Pattern 2

ENDTEST
