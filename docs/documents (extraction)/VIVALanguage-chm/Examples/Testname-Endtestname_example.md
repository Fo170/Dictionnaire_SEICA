# TESTNAME-ENDTESTNAME 
 example

! Template cluster

DECLARE CHANNEL RESET=2;

DECLARE CHANNEL CLK=1;

DECLARE RUNTIME FLOAT VAL=1.3;

DECLARE RUNTIME STRING CODENAME,PARTNAME,PARTVALUE,MSGPASS,MSGFAIL;

DECLARE RUNTIME FLOAT LOWVALUE,HIGHVALUE,MEASVALUE,TOLVALUE;

START TEMPLA STATIC;

TESTNAME=TEST1;

~CALC PARTNAME="R1";

~CALC CODENAME="R1_1K_1/8W_2%";

~CALC PARTVALUE="1000";

SECTION USER;

~WRITE "CHECK R1 LEADS";

SECTION SET;

~SET PW N=1 V=6 I=2 ;

~SET IMM MEAS_V=AUTO MEAS_I=AUTO INP_VNEG=L1 REF_I=L3 ;

~SET AGND OUT=L1 ;

~LI1=RESET;

SECTION USER;

~WRITE "USER SECTION";

SECTION DIGITAL;

IH reset;

SECTION USER;

/;

! INIT THE BOARD

SECTION MEAS;

~MEAS IMM INTO=AR MEAS=AVERAGE_V ;

SECTION USER;

! MEAS RESISTER

 SECTION 
 DIAG;

~CALC HIGHVALUE=1010;

~CALC LOWVALUE=990;

~CALC MEASVALUE=1000;

~ATEST AR HI=HIGHVALUE LO=LOWVALUE ;

SECTION USER;

! END OF TEST

SECTION END;

ENDTESTNAME;

TESTNAME=RESET;

~CALC PARTNAME="ABCDER";

SECTION USER;

~WRITE "PAOLO";SECTION SET;

~SET DRB V=VAL OUT=L2 ;

~SET DRA V=1.3 RANGE=1V FREQ=(AR) WAVE=SINE OUT=L1 ;

~SET IMM MEAS_V=AUTO MEAS_I=AUTO FREQ=1000 INP_VPOS=L1 INP_VNEG=L3 ;

~SET AGND OUT=L3 ;

~FLI1=CLK;

~FLI2=RESET;

SECTION DIGITAL;

ih CLK;SECTION USER;

~PAUSE;SECTION MEAS;

~MEAS IMM INTO=VAL MEAS=MAX_V ;

SECTION DIAG;

~CALC HIGHVALUE=1;

~CALC LOWVALUE=0.1;

~CALC MEASVALUE=0.5;

~ATEST AR HI=HIGHVALUE LO=LOWVALUE ;

SECTION END;

ENDTESTNAME;

ENDTEST;
