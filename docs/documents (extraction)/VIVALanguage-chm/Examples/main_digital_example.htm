MAIN DIGITAL EXAMPLE

DECLARE RUNTIME INTEGER CAN;

DECLARE CHANNEL CS=125;

DECLARE CHANNEL WR=23;

DECLARE GROUP DATA=1,2,3,4;

 

START PRG1;

DYNAMIC TIMING T1;

IH CS; ! CHANNEL 125 IS DRIVING HIGH AND CHECKED TO BE HIGH AT THE END OF THE PATTERN

/;

IH 23; ! CHANNEL  125,23 ARE DRIVING HIGH

/;

IL 2,4,6,8;  ! CHANNEL  125,23 ARE DRIVING HIGH  2,4,6,8 are driving low

/;

MH 1/8;   ! CHANNEL 125,23 IS DRIVING HIGH AND CHECKED TO BE HIGH AT THE END OF THE PATTERN  

              ! CHANNEL 1,2,3,4,5,6,7,8 ARE DRIVING HIGH AND IGNORED STATE AT END OF THE PATTERN

/;

OM   DATA; ! CHANNEL 125,23 IS DRIVING HIGH AND CHECKED TO BE HIGH AT THE END OF THE PATTERN

                 ! CHANNEL 5,6,7,8 ARE DRIVING HIGH AND IGNORED STATE AT END OF THE PATTERN

                 ! CHANNEL 1,2,3,4 ARE TRISTATED AND IGNORED STATE

 

/;

OH 1 ;          ! CHANNEL 1 ARE SENSED TO BE LOW AT STROBE TIME;

OL 2 ;          ! CHANNEL 2 ARE SENSED TO BE HIGH AT STROBE TIME;

OM 5,6,7,8;   ! CHANNEL 125,23 IS DRIVING HIGH AND CHECKED TO BE HIGH AT THE END OF THE PATTERN

/;                 ! CHANNEL 3,4, ARE TRISTATED AND IGNORED STATE  

 

STATIC;

 

ENDTEST;
