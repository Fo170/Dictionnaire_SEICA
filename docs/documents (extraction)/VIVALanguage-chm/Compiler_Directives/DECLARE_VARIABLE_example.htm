DECLARE VARIABLE example

 

Example 1

 

DECLARE VARIABLE A,B,IN1; Variable declarations

DECLARE VARIABLE J=5; Declaration and initialization

START PRVAR;

@CALC A=2;

@CALC B=1;

@CALC IN1=15;

/;

@CALC A=A+B/J;

@CALC IN1=IN1-A;

ENDTEST;
