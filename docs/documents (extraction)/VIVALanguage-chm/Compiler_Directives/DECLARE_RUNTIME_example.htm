DECLARE RUNTIME example

 

Example 1

 

DECLARE RUNTIME INTEGER A

DECLARE RUNTIME INTEGER B,J,IN1;

DECLARE RUNTIME FLOAT TEMP,CUR;

DECLARE RUNTIME STRING NOME;

START PRG1 STATIC;

~CALC A=3;

~CALC B=5;

~CALC TEMP=14.7;

~CALC CUR=150;

~CALC NOME=“CHANNEL nr. 1”;

/;

ENDTEST;
