~READ_TIMER example

 

Example 1

 

FLOAT F1,F2,F3;

integer j=0,K;

subr mytimer ;

writel on=4 "TIMER END";

j=1;  !! TO BREAK THE LOOP

RETURN;

endsubr;

declare subroutine  mytimer;

START PROTIMER STATIC;

ON_TIMER CALL MYTIMER ;

READ TIMER1 AR;  !! TIME IN MICRO SINCE START PROGRAM

WRITEL   "TIMEFROM START=" AR;

READ TIMER0 AR;  !! TIME IN MICRO SINCE START PROGRAM

AR=AR/1000000.;

WRITEL   "TIMEFROM PC POWER (SEC) ON=" AR;

TIMER 1 CLEAR;

TIMER 2 CLEAR;

READ TIMER1 AR;  !! TIME IN MICRO SINCE LAST CLEAR;

WRITEL   "TIMEFROM (USEC) CLEAR=" AR;

READ TIMER3 AR;  !! TIME IN MICRO SINCE LAST CLEAR;

WRITEL   "TIMEFROM START=" AR;

READ TIMER2 F1;

READ TIMER2 F2;

F3=F2-F1;

WRITEL   "TIMEBEEWEEN 2 READ TIMER " F3 ;

READ TIMER2 F1;

DELAY 100M;

!WRITEL "AAAA";

!AR=AR+1;

READ TIMER2 F2;

AR=F2-F1-F3;

WRITEL   "TIME FOR 1 ISTRUCTION " AR ;

TIMER 2 CLEAR;

READ TIMER2 STATUS F1;

WRITEL on=1  "TIMER 2 STATUS 0=CLEAR= " F1 ;

TIMER 2 ARM VAL=1.;

READ TIMER2 F1;

WRITEL on=1   "TIME VALUE SET TO 1 SEC " F1 ;

~DELAY 600M;

READ TIMER2 STATUS F1;

WRITEL on=1  "TIMER 2 STATUS= 1=ARMED= " F1 ;

~DELAY 300M;

READ TIMER2 F1;

WRITEL on=1  "TIME VALUE SET TO 1 SEC " F1 ;

~DELAY 200M;

READ TIMER2 STATUS F1;

WRITEL on=1  "TIMER 2 STATUS 2=ENDED= " F1 ;

j=0;

K=0;

TIMER 2 ARM  VAL=20.;

while j=0;

K=K+1;

READ TIMER2 F1;

WRITEL on=1  USING="A20,F12.0,A4,I4,A4","\CU\001\001 TIMER2= " F1,"   ", K, "   " ;

!F2=-3000000;

!IF (F1>F2) ;

!TIMER 2 STOP;

!ENDIF;

endwhile;

READ TIMER2 F1;

!CALL MYTIMER;

~PAUSE AR;

ENDTEST;
