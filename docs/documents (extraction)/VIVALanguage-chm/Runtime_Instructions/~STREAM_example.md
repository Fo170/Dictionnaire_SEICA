# ~STREAM example

Example 1

START PRG1 STATIC;

~WRITE "AAAAAAA"; !! 
 DEFAULT WRITE STREAM ONLY TO THE TERMINAL

~STREAM DIAG 
 ;

~WRITE "BBBBBBB"; !! WRITE WILL BE 
 OUTPUTTING ONLY TO THE DIAGNOSTIC REPORT

~FILE3 OPEN_WRITE "FILEC.DAT";

~STREAM TERM 
 FILE3 ; ! 
 DETACH ALL AND ATTACH FILE3 AND TERMINAL

~WRITE "CCCCCCC"; !! WRITE WILL BE 
 OUTPUTTING TO FILEC.DAT AND TO THE TERMINAL

~FILE4 OPEN_WRITE "FILED.DAT";

~STREAM ATTACH 
 FILE4 ; ! 
 ATTACH ALSO FILE4 

~WRITE "DDDDDDDD"; !! 
 WRITE WILL BE OUTPUTTING TO FILEC.DAT , FILED.DAT AND TO THE TERMINAL

~STREAM DETACH 
 FILE3 ; ! 
 DETACH FILE4 

~WRITE "EEEEEEEE"; !! WRITE WILL BE 
 OUTPUTTING TO FILEC.DAT AND TO THE TERMINAL

~FILE4 CLOSE;

~FILE3 CLOSE;

ENDTEST;

AFTER EXECUTION:

TERMINAL WILL DISPLAY

AAAAAA

CCCCCC

DDDDDD

EEEEEEE

REPORT WILL 
 SHOW UP

BBBBBBBB

FILE DATAC.DAT WILL CONTAIN

CCCCCC

DDDDDD

FILE DATAD.DAT WILL CONTAIN

DDDDDD

EEEEEEE
