~DELAY example

 

Example  1

 

DECLARE RUNTIME INTEGER NCAN, SUBERR, MAXERR;

START PRG1 STATIC;

/;

~DO NCAN;

~CURSOR 60,3;

~WRITE "CHANNEL N. " CAN;

~CALC CAN=CAN+1;

~COMP SUBERR HI=MAXERR ONERROR FINE;

~DELAY 1M;

~DONE;

LABEL=FINE;

/;

ENDTEST;
